-- 1. Create Properties Catalog Table
CREATE TABLE public.properties (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    beds INT NOT NULL,
    price INT NOT NULL,
    style TEXT NOT NULL,
    amenities TEXT[] DEFAULT '{}'::TEXT[],
    description TEXT
);

-- 2. Create Tour Bookings Table
CREATE TABLE public.tour_bookings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    property_id TEXT REFERENCES public.properties(id),
    full_name TEXT NOT NULL,
    email TEXT NOT NULL,
    phone_number TEXT NOT NULL,
    preferred_date TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 1. Create an ENUM type for booking statuses
CREATE TYPE booking_status AS ENUM ('pending', 'confirmed', 'cancelled', 'rescheduled');

-- 2. Add the status column to tour_bookings with a default value
ALTER TABLE public.tour_bookings 
ADD COLUMN status booking_status DEFAULT 'confirmed'::booking_status;

-- 3. Add an updated_at timestamp hook for structural tracking
ALTER TABLE public.tour_bookings 
ADD COLUMN updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL;
